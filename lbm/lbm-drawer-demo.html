$('.topDrawer').eq(0).appendTo('body').drawer({
	  direction:'top',   //抽屉的位置：top/bottom/left/right
	  width:'100%',    //抽屉的宽度
	  height:'50%',     //抽屉的高度
	  openBtn:'openTopBtn',   //打开抽屉的按钮的样式名
	  closeBtn:'mui-btn',   //关闭抽屉的按钮的样式名
	  eventType:'click',    //触发事件设置，默认是click
	  cb:function(){}    //回调函数
});
